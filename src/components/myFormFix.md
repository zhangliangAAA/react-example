class KInput extends Component {
  render() {
    const {name, onChange, value, type, prefix, onFocus} = this.props;
    return (
      <div>
        {prefix}
        <input name={name} type={type} onChange={onChange} value={value}
        onFocus={onFocus}/>
      </div>
    );
  }
}
